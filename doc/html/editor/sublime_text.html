    常用快捷

    Windows Linux Mac
    column mode [Ctrl] ^ [Alt] [shift] ^ [control]
    command palette [Ctrl] ^ [Shift] + p [shift] ^ [command] + p
    show/hide side [Ctrl] + k, [Ctrl] + b [command] + k, [command] + b
    show console [Ctrl] + ` [control] + `
    goto anything/file [Ctrl] + p [command] + p
    goto symbol/function [Ctrl] + r [command] + r
    goto line [Ctrl] + g [control] + g
    goto next find [Ctrl] + i [command] + i
    goto prev find [Ctrl] ^ [Shift] + i [command] ^ [shift] + i
    goto next file [Ctrl] + PageDown [command] ^ [shift] + ]
    goto prev file [Ctrl] + PageDown [command] ^ [shift] + [
    goto next in stack [Ctrl] + [Tab]
    goto prev in stack [Ctrl] ^ [Shift] + [Tab]
    folding close [Ctrl] ^ [Shift] + [ [command] ^ [option] + [
    folding open [Ctrl] ^ [Shift] + ] [command] ^ [option] + ]
    folding level n (n = 1 ~ 9) [Ctrl] + k, [Ctrl] + n
